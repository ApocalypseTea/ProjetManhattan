SELECT R.target, json_object('target', R.target, 'pays', R.value, 'nbRequetes', R2.value)
    FROM record AS R
    LEFT JOIN record AS R2 ON R.target = R2.target  AND R2.propertyName = 'NbRequetes'
    WHERE R.propertyName = 'OrigineGeographique' AND
          R.target = @Target AND
          R.date BETWEEN @DateDebut AND @DateFin;
